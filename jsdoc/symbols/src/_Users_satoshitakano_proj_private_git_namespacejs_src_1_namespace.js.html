 42 var FunctionPrototype = function() {
 43 	var self = this;
 44 	// util
 45 	function getMethodName(method) {
 46 		var name = method.name;
 47 		if (name == undefined) {
 48 			name = /function\s*(.*)\s*\(/mgi.exec(method)[1];
 49 			if (name == null) return;
 50 		}
 51 		return name;
 52 	}
 53 	this.getMethodName = getMethodName;
 54 	
 55 	// hierarchie util
 56 	function wrap(key, method) {
 57 		var wrapper = method;
 58 		// use by $super() method
 59 		wrapper.superMethod = self.substance.prototype[key];
 60 		return wrapper;
 61 	};
 62 	function $super() {
 63 		return arguments.callee.caller.superMethod.apply(this, arguments);
 64 	};
 65 	
 66 	/**
 67 	* クラスからインスタンスを得ます.<br/>
 68 	* 引数に渡したオブジェクトは,クラスのコンストラクタにそのまま引き渡されます.
 69 	**/
 70 	this.gen = function() {
 71 		var obj = new this();
 72 		if (obj.initialize != undefined) {
 73 			obj.initialize.apply(obj, arguments);
 74 		}
 75 		return obj;
 76 	};
 77 	
 78 	/**
 79 	* クラスからシングルトンインスタンスを得ます.<br/>
 80 	* 引数に渡したオブジェクトは,クラスのコンストラクタにそのまま引き渡されます.
 81 	**/
 82 	this.getInstance = function() {
 83 		var instance = this.instance;
 84 		if (!instance) {
 85 			instance = this.instance = new this();
 86 			if (instance.initialize != undefined) {
 87 				instance.initialize.apply(instance, arguments)
 88 			}
 89 		}
 90 		return this.instance;
 91 	}
 92 	
 93 	/**
 94 	* コンストラクタを定義します.
 95 	* @param initialize コンストラクト時に呼ばれるクロージャ
 96 	**/
 97 	this.init = function(initialize) {
 98 		self.substance.prototype.initialize = wrap("initialize", initialize);
 99 	};
100 	
101 	/**
102 	* 継承元を指定します.
103 	* @param obj Namespace に定義されている、Singleton ではないクラス(非インスタンス)を渡します.
104 	**/
105 	this.ex = function(obj) {
106 		var proto = obj.gen();
107 		self.substance.superClass = obj.prototype;
108 		self.substance.prototype = proto;
109 		self.substance.prototype.$super = $super;
110 	};
111 	
112 	/**
113 	* オブジェクトをモジュールとして mixin します.<br/>
114 	* @param {Object} module include したいオブジェクト. これは JavaScript native Object でも可です.
115 	* @param {Boolean} overwrite true が渡された場合、既に存在する同名のメンバを上書きします.
116 	**/
117 	this.include = function(module, overwrite) {
118 		for (var p in module) {
119 			if (overwrite || this[p] == undefined) {
120 				this[p] = module[p];
121 			}
122 		}
123 	};
124 	
125 	/**
126 	* getter メソッドを定義します.<br/>
127 	* getter に対応していない環境では定義は行われません.
128 	* @param name {String} getter名
129 	* @param func getterが呼び出された時に呼ばれるクロージャ.
130 	**/
131 	this.getter = function (name, func) {
132 		var p = self.substance.prototype;
133 		if ("__defineGetter__" in p) {
134 			p.__defineGetter__(name, func);
135 		}
136 	}
137 	
138 	/**
139 	* setter メソッドを定義します.<br/>
140 	* setter に対応していない環境では定義は行われません.
141 	* @param name {String} setter名
142 	* @param func setterが呼び出された時に呼ばれるクロージャ.
143 	**/
144 	this.setter = function (name, func) {
145 		var p = self.substance.prototype;
146 		if ("__defineSetter__" in p) {
147 			p.__defineSetter__(name, func);
148 		}
149 	}
150 	
151 	/**
152 	* メソッドを定義します.
153 	* @param method {Function} 名前付き関数.この関数の名前がメソッド名になります.
154 	**/
155 	this.def = function(method){
156 		var name = getMethodName(method);
157 		self.substance.prototype[name] = wrap(name, method);
158 	};
159 	this.defInObj = function(method) {
160 		var name = getMethodName(method);
161 		this[name] = method;
162 	}
163 }
164 FunctionPrototype = new FunctionPrototype();

167 var InternalNamespacePrototype = function() {
168 	this.nsName = "";
169 	var self = this;
170 	
171 	/**
172 	* @memberOf Namespace.prototype
173 	* @description この名前空間のスコープを使用します.
174 	* @param func クロージャ
175 	*/
176 	this.use = function(func) {
177 		self = this;
178 		var oldDef = global.proto;
179 		var oldSingleton = global.singleton;
180 		global.proto = this.proto;
181 		global.singleton = this.singleton;
182 		func.call(this);
183 		global.proto = oldDef;
184 		global.singleton = oldSingleton;
185 	};
186 	
187 	/**
188 	* @memberOf Namespace.prototype
189 	* @description Class (プロトタイプ)を定義します.
190 	* @param namedFunc 名前付き関数. この関数名がクラス名になります.
191 	* @description このメソッドに渡す名前付き関数のスコープでのみ有効になる,<br/>
192 	*  ex, include, init, def, getter, setter, $$ <br/>
193 	* というグローバル関数・変数があります.<br/>
194 	* それぞれの説明は下記を参照してください.<br/><br/>
195 	* ex : {@link ex}<br/>
196 	* include : {@link include}<br/>
197 	* init : {@link init}<br/>
198 	* def : {@link def}<br/>
199 	* getter : {@link getter}<br/>
200 	* setter : {@link setter}<br/>
201 	* $$ : クラスへの参照. $$ に対して def を呼び出すとクラスメソッド,<br/>
202 	* $$.hoge = "hoge" とするとクラス変数になります.
203 	*/
204 	this.proto = function(namedFunc) {
205 		var tmpSelf = self;
206 		if (this instanceof Namespace) {
207 			tmpSelf = this;
208 		}
209 		var name = FunctionPrototype.getMethodName(namedFunc);
210 		if (tmpSelf[name] != undefined) alert("Warning: " + tmpSelf.nsName + "'s " + name + " was overwritten.");
211 		if (!global.debug)
212 			tmpSelf[name] = function() {};
213 		else
214 			tmpSelf[name] = eval("(function " + name + " () {})");
215 		
216 		if (tmpSelf[name] == undefined) tmpSelf[name] = new Function();
217 		var proto = tmpSelf[name];
218 		proto.gen = FunctionPrototype.gen;
219 		proto.def = FunctionPrototype.defInObj;
220 		proto.prototype.include = FunctionPrototype.include;
221 		proto.prototype.$class = proto;
222 		
223 		// fake dynamic scope
224 		var old$$ = global.$$;
225 		var oldInit = global.init;
226 		var oldEx = global.ex;
227 		var oldMeth = global.def;
228 		var oldGetter = global.getter;
229 		var oldSetter = global.setter;
230 		FunctionPrototype.substance = proto;
231 		
232 		global.$$ = proto;
233 		global.init = namedFunc.init = FunctionPrototype.init;
234 		global.ex = namedFunc.ex = FunctionPrototype.ex;
235 		global.def = namedFunc.def = FunctionPrototype.def;
236 		global.getter = FunctionPrototype.getter;
237 		global.setter = FunctionPrototype.setter;
238 		namedFunc.call(namedFunc);
239 
240 		global.$$ = old$$;
241 		global.init = oldInit;
242 		global.ex = oldEx;
243 		global.def = oldMeth;
244 		global.getter = oldGetter;
245 		global.setter = oldSetter;
246 		
247 		proto.prototype.proto = proto;
248 	};
249 	
250 	/**
251 	* @memberOf Namespace.prototype
252 	* @description Singleton Class (プロトタイプ)を定義します.
253 	* @param namedFunc 名前付き関数. この関数名がクラス名になります.
254 	* @description このメソッドに渡す名前付き関数のスコープでのみ有効になる,<br/>
255 	*  ex, include, init, def, getter, setter, $$ <br/>
256 	* というグローバル関数・変数があります.<br/>
257 	* それぞれの説明は下記を参照してください.<br/><br/>
258 	* ex : {@link ex}<br/>
259 	* include : {@link include}<br/>
260 	* init : {@link init}<br/>
261 	* def : {@link def}<br/>
262 	* getter : {@link getter}<br/>
263 	* setter : {@link setter}<br/>
264 	* $$ : クラスへの参照. $$ に対して def を呼び出すとクラスメソッド,<br/>
265 	* $$.hoge = "hoge" とするとクラス変数になります.
266 	*/
267 	this.singleton = function(namedFunc) {
268 		var tmpSelf = self;
269 		if (this instanceof Namespace) {
270 			tmpSelf = this;
271 		}
272 		var name = FunctionPrototype.getMethodName(namedFunc);
273 		if (tmpSelf[name] != undefined) alert("Warning: " + tmpSelf.nsName + "'s " + name + " was overwritten.");
274 		
275 		tmpSelf[name] = eval("(function " + name + " () {})");
276 		if (tmpSelf[name] == undefined) tmpSelf[name] = new Function();
277 		var proto = tmpSelf[name];
278 		proto.getInstance = FunctionPrototype.getInstance;
279 		proto.def = FunctionPrototype.defInObj;
280 		proto.prototype.include = FunctionPrototype.include;
281 		proto.prototype.$class = proto;
282 		
283 		// fake dynamic scope
284 		var old$ = global.$;
285 		var old$$ = global.$$;
286 		var oldInit = global.init;
287 		var oldEx = global.ex;
288 		var oldMeth = global.def;
289 		var oldGetter = global.getter;
290 		var oldSetter = global.setter;
291 		FunctionPrototype.substance = proto;
292 		
293 		global.$ = namedFunc;
294 		global.$$ = proto;
295 		global.init = namedFunc.init = FunctionPrototype.init;
296 		global.ex = namedFunc.ex = FunctionPrototype.ex;
297 		global.def = namedFunc.def = FunctionPrototype.def;
298 		global.getter = FunctionPrototype.getter;
299 		global.setter = FunctionPrototype.setter;
300 		namedFunc.call(namedFunc);
301 
302 		global.$ = old$; 
303 		global.$$ = old$$;
304 		global.init = oldInit;
305 		global.ex = oldEx;
306 		global.def = oldMeth;
307 		global.getter = oldGetter;
308 		global.setter = oldSetter;
309 		
310 		proto.prototype.proto = proto;
311 	};
312 }

316 /**
317 * @class all classes are defined to Namespace class.
318 * @param str 名前空間を String 型で指定します. (例: "jp.example.hoge")
319 * @description "jp.example.hoge" を引数に Namespace を new した場合,<br/>
320 * jp, jp.example, jp.example.hoge というオブジェクトが作成されます.<br/>
321 * すでに存在する場合は新しく作成されることはありません.
322 */
323 Namespace = function(str) {
324 	var ns = str.split('.');
325 	
326 	var here = global;
327 	if (ns.length == 1) {
328 		if (here[str] != undefined)
329 			return here[str];
330 		
331 		here[str] = this;
332 		this.nsName = str;
333 		return this;
334 	}
335 	
336 	var name = "";
337 	for (var i = 0, l = ns.length, ll = l-1; i < l; i++) {
338 		if (typeof(here[ns[i]]) == "undefined") {
339 			here[ns[i]] = new Namespace(ns[i]);
340 		}
341 		if (i > 0) name += ".";
342 		name += ns[i];
343 		here[ns[i]].nsName = name;
344 		/* else {
345 			// 他のライブラリ等ですでに使われてしまっている場合は mix
346 		}*/
347 		here = here[ns[i]];
348 	}
349 	return here;
350 }
351 Namespace.prototype = internalNamespacePrototype;

355 	var ns = this;
356 	
357 	 /**
358 	 * @class bind object with function
359 	 */
360 	 proto(function Performer() {
361 	 	/**
362 	 	* @constructs
363 	 	* @param target オブジェクト
364 	 	* @param func 関数
365 	 	* @param func の引数
366 	 	*/
367 	 	init(function(target, func, args) {
368 	 		this.target = target;
369 	 		this.func = func;
370 	 		this.args = args;
371 	 	});
372 	 	
373 	 	/**
374 	 	* func を実行します.
375 	 	*/
376 	 	def(function perform() {
377 	 		this.func.apply(this.target, arguments);
378 	 	})
379 	 });
380 	
381 	/** 
382 	 * @private
383 	 */
384 	proto(function Utilitie() {
385 		def(function listen(target, type, func) {
386 			if (target.addEventListener)
387 				target.addEventListener(type, function (e) {
388 					func.call(this, e);
389 				}, false);
390 			else if (target.attachEvent)
391 				target.attachEvent(type, function (e) {
392 					func.call(this, e);
393 				});
394 		});
395 		
396 		def(function unlisten(target, type, func) {
397 			if (target.removeEventListener)
398 				target.removeEventListener(type, func);
399 			else if (target.attachEvent)
400 				target.detachEvent(type, func);
401 		})		
402 	});
403 	
404 	/**
405 	* @class メインクラス.
406 	* 
407 	**/
408 	singleton(function Main() {
409 		// To initialize when the Main.gen(params) called.
410 		init(function() {
411 				var self = this;
412 				var util = ns.Utilitie.gen();
413 				var isIE = navigator.userAgent.toLowerCase().indexOf("msie") != -1;
414 				
415 				util.listen(global, isIE ? "onload" : "load", function () {
416 						util.unlisten(global, isIE ? "onload" : "load", arguments.callee);
417 						if (self.runner) self.runner.call(global);
418 				});
419 		})
420 		
421 		/**
422 		* アプリケーションの初期化処理が終わった時に,引数に渡されたクロージャが実行されます.
423 		*/
424 		def(function main(runner) {
425 			this.runner = runner;
426 		})
427 	})
