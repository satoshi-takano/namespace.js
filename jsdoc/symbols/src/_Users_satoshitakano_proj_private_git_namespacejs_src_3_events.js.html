 29 new Namespace(namespace_lib_events).use(function() {
 30 	var ns = this;
 31 	// dynamical creating a internal namespace
 32 	var nsi = new Namespace(ns.nsName + ".internal");
 33 	var userAgent = new Namespace(namespace_lib_platform).browser.UserAgent.gen();
 34 	
 35 	/* internal */
 36 	nsi.proto(function EventTargetSet(listener) {
 37 		init(function (listener) {
 38 			this.listener = listener;
 39 		})
 40 	});

 43 	/** 
 44 	* @class イベントを発行するための機能を定義しています.
 45 	*/
 46 	proto(function EventDispatcher() {
 47 		/** 
 48 		* @memberOf EventDispatcher 
 49 		* @param 
 50 		*/
 51 		init(function (target) {
 52 			this.observers = {};
 53 			this.target = target;
 54 		});
 55 		
 56 		/**
 57 		* イベントリスナを追加します.
 58 		* @param type {String} イベントタイプ.
 59 		* @param callback {Function} イベント発生時に評価する関数.
 60 		*/
 61 		def(function addEventListener(type, callback) {
 62 			var list = this.observers[type];
 63 			var flg = false;
 64 			if (list) {
 65 				var numTargets = list.length;
 66 				for (var i = 0; i < numTargets; i++)
 67 					if (list[i] == callback)
 68 						flg = true;
 69 				if (!flg)
 70 					list.push(callback);
 71 			} else {
 72 				this.observers[type] = [];
 73 				this.observers[type].push(callback);
 74 			}
 75 			return flg;
 76 		});
 77 		
 78 		/** 
 79 		* addEventListener のエイリアス. 
 80 		* @deprecated 将来的に廃止される可能性があります.
 81 		*/
 82 		def(function ae(type, callback) {
 83 			this.addEventListener(type, callback);
 84 		});
 85 		
 86 		/**
 87 		* イベントリスナを削除します.
 88 		* @param type {String} イベントタイプ.
 89 		* @param callback {Function} 登録済み関数.
 90 		*/
 91 		def(function removeEventListener(type, callback) {
 92 			var list = this.observers[type];
 93 			var numTargets = list.length;
 94 			for (var i = 0; i < numTargets; i++) {
 95 				if (list[i] && list[i] == callback) {
 96 					list.splice(i, 1);
 97 					return;
 98 				}
 99 			}
100 		});
101 
102 		/** 
103 		* removeEventListener のエイリアス. 
104 		* @deprecated 将来的に廃止される可能性があります.
105 		*/
106 		def(function re(type, performer) {
107 			this.removeEventListener(type, performer);
108 		});
109 		
110 		def(function dispatchEvent(event) {
111 			var list = this.observers[event.type];
112 			if (list) {
113 				var numTargets = list.length;
114 				for (var i = 0; i < numTargets; i++) {
115 					list[i].call(this, event);
116 				}
117 			}
118 		});
119 	});

123 	* @class DOMに関するイベントを定義しています.
124 	*/
125 	proto(function DOMEvent() {
126 		var isIE = userAgent.isIE();
127 		if (isIE) {
128 			/** @memberOf DOMEvent */
129 			$$.INIT = "onload";
130 			/** @memberOf DOMEvent */
131 			$$.LOAD = "onload";
132 		} else {
133 			$$.INIT = "load";
134 			$$.LOAD = "load";
135 		}
136 	});

140 	* @class DOMのマウス操作に関するイベントを定義しています.
141 	*/

143 		var isIE = userAgent.isIE();
144 		if (isIE) {
145 			/** @memberOf DOMMouseEvent */
146 			$$.CLICK = "onclick";
147 			/** @memberOf DOMEvent */
148 			$$.MOUSE_DOWN = "onmousedown";
149 			/** @memberOf DOMEvent */
150 			$$.MOUSE_MOVE = "onmousemove";
151 			/** @memberOf DOMEvent */
152 			$$.MOUSE_OUT = "onmouseout";
153 			/** @memberOf DOMEvent */
154 			$$.MOUSE_OVER = "onmouseover";
155 			/** @memberOf DOMEvent */
156 			$$.MOUSE_UP = "onmouseup";
157 			/** @memberOf DOMEvent */
158 			$$.MOSUE_WHEEL = "onmousewheel";
159 		} else {
160 			$$.CLICK ="click";
161 			$$.MOUSE_DOWN = "mousedown";
162 			$$.MOUSE_MOVE = "mousemove";
163 			$$.MOUSE_OUT = "mouseout";
164 			$$.MOUSE_OVER = "mouseover";
165 			$$.MOUSE_UP = "mouseup";
166 			$$.MOSUE_WHEEL = "mousewheel";
167 		}
168 		
169 	});

171 	/** 
172 	* @class FLEvent
173 	*/
174 	proto(function FLEvent() {
175 		/**
176 		* @memberOf FLEvent
177 		* @param type {String} イベントタイプ
178 		* @param caller {Object} イベント発行元オブジェクト
179 		* @param origin {Event} JavaScript 実行環境ネイティブの Event オブジェクト
180 		*/
181 		init(function (type, caller, origin) {
182 			this.type = type;
183 			this.origin = origin;
184 			this.currentTarget = caller;
185 		});
186 		
187 		/**
188 		* デフォルトの処理実行をキャンセルします.
189 		*/
190 		def(function preventDefault() {
191 			if (this.origin.preventDefault)
192 				this.origin.preventDefault();
193 			else
194 				this.origin.returnValue = false;
195 		});
196 		
197 		/**
198 		* イベントの伝搬を停止します.
199 		*/
200 		def(function stopPropagation() {
201 			if (this.origin.stopPropagation)
202 				this.origin.stopPropagation();
203 			else
204 				this.origin.cancelBubbles = true;
205 		});
206 		
207 		/** @memberOf FLEvent */
208 		$$.APP_LAUNCH = "appLaunch";
209 		/** @memberOf FLEvent */
210 		$$.COMPLETE = "complete";
211 	});
