 48 		* 引数に 16進数での色の値を与えて Color オブジェクトを作ります.
 49 		* @param {Number}
 50 		* @memberOf Color

 52 		init(function(col16) {
 53 			this.color = col16;

 56 		/** red の値. [read-write] */
 57 		getter("r", function() {
 58 			return this.color >> 16;
 59 		})
 60 		setter("r", function(val) {
 61 			this.color = val << 16 | this.g << 8 | this.b;
 62 		})
 63 		
 64 		/** green の値. [read-write] */
 65 		getter("g", function() {
 66 			return this.color >> 8 & 0xff;
 67 		})
 68 		setter("g", function(val) {
 69 			this.color = this.r << 16 | this.b << 8 | this.b;
 70 		})
 71 		
 72 		/** blue の値. [read-write] */
 73 		getter("b", function() {
 74 			return this.color & 0xff;
 75 		})
 76 		setter("b", function(val) {
 77 			this.color = this.r << 16 | this.g << 8 | val;
 78 		})
 79 	})

531 			this.numChildren.times(function (i) {

545 			var numChildren = this.numChildren;
546 			numChildren.times(function (i) {
547 				if (this.children[i] == child) this.children.splice(i, 1);
548 			}, this);
549 			var left = this.children.splice(0, index);
550 			var right = this.children.splice(0, this.numChildren);
551 			left.push(child);
552 			this.children = left.concat(right);
553 			child.parent = this;
554 			child._stg = this._stg;
555 		})

562 			var i = 0;
563 			this.children.each(function (c) {
564 				if (c == child) {
565 					this.children.splice(i, 1);
566 					return 0;
567 				}
568 				i++;
569 			}, this);
570 		})
