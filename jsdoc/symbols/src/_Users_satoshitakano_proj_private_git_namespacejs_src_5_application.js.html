 26 new Namespace(namespace_lib_app).use(function() {
 27 	var ns = this;
 28 		
 29 	 /** 
 30 	 * @class Application
 31 	 * @augments EventDispatcher
 32 	 */
 33 	singleton(function Application() {
 34 		var nse = new Namespace(namespace_lib_events);
 35 		ex(nse.EventDispatcher);
 36 		
 37 		init(function() {
 38 			var self = this;
 39 			/** 
 40 			* 表示領域内でのマウスポインタの x 位置
 41 			* @memberOf Application.prototype
 42 			*/
 43 			this.mouseX = 0;
 44 			/** 
 45 			* 表示領域内でのマウスポインタの y 位置
 46 			* @memberOf Application.prototype 
 47 			*/
 48 			this.mouseY = 0;
 49 			/** @memberOf Application.prototype */
 50 			this.userAgent = new Namespace(namespace_lib_platform).browser.UserAgent.gen();
 51 			
 52 			var nscore = (new Namespace(namespace_lib_core));
 53 			var util = nscore.Utilitie.gen();
 54 			var isIE = this.userAgent.isIE();
 55 			util.listen(document, isIE ? "onmousemove" : "mousemove", function(mouseMove) {
 56 				if (!isIE) {
 57 					this.mouseX = mouseMove.clientX;
 58 					this.mouseY = mouseMove.clientY;
 59 					if (document.body)	{ this.mouseY +=  document.body.scrollTop;}
 60 				} else {
 61 					this.mouseX = mouseMove.clientX;
 62 					this.mouseY = mouseMove.clientY + document.documentElement.scrollTop;
 63 				}
 64 			});
 65 			
 66 		});
 67 	});
 68 
 69 });
