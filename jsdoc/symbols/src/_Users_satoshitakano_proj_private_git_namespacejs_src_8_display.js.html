 25 /**
 26 * @fileOverview DOM の View に関係するオブジェクトが定義されています.

 29 new Namespace(namespace_lib_display).use(function() {
 30 	var ns = this;
 31 	var nsc = new Namespace(namespace_lib_core);
 32 	var nse = new Namespace(namespace_lib_events);
 33 	var nsg = new Namespace(namespace_lib_geom);
 34 	var app = (new Namespace(namespace_lib_app)).Application.getInstance();
 35 	
 36 	/** 
 37 	* @class DOM の表示オブジェクトように高レベルAPIを提供します.
 38 	* @augments EventDispatcher
 39 	*/
 40 	proto(function DisplayObject() {
 41 		ex(nse.EventDispatcher);
 42 		
 43 		/**
 44 		* DOM オブジェクトをを引数に、DisplayObject オブジェクトを作成します.
 45 		* @param {Object} DOM
 46 		* @memberOf DisplayObject
 47 		*/
 48 		init(function(domElement) {
 49 			this.domElement = domElement;
 50 			this.parent = null;
 51 			this.alpha = 1;
 52 		});
 53 		
 54 		/**
 55 		* x 座標を返します.
 56 		* @return {Number}
 57 		*/
 58 		def(function getX() {
 59 			return this.domElement.style.left.replace("px", "") || 0;
 60 		});
 61 		
 62 		/**
 63 		* x 座標を指定します.
 64 		*/
 65 		def(function setX(val) {
 66 			this.domElement.style.marginLeft = val + "px";
 67 		});
 68 		
 69 		/**
 70 		* y 座標を返します.
 71 		* @return {Number}
 72 		*/
 73 		def(function getY() {
 74 			return this.domElement.style.top.replace("px", "") || 0;
 75 		});
 76 		
 77 		/**
 78 		* y 座標を指定します.
 79 		*/
 80 		def(function setY(val) {
 81 			this.domElement.style.top = val + "px";
 82 		});
 83 		
 84 		/**
 85 		* 幅を返します.
 86 		* @return {Number}
 87 		*/
 88 		def(function getWidth() {
 89 			return this.domElement.offsetWidth;
 90 		});
 91 		
 92 		/**
 93 		* 幅を指定します.
 94 		*/
 95 		def(function setWidth(val) {
 96 			this.domElement.style.width = val + "px";
 97 		});
 98 		
 99 		/**
100 		* 高さを返します.
101 		* @return {Number}
102 		*/
103 		def(function getHeight() {
104 			return this.domElement.offsetHeight;
105 		});
106 		
107 		/**
108 		* 高さを指定します.
109 		*/
110 		def(function setHeight(val) {
111 			this.domElement.style.height = val + "px";
112 		});
113 		
114 		/**
115 		* mouse x 座標を返します.
116 		* @return {Number}
117 		*/
118 		def(function getMouseX() {
119 			return app.mouseX - this.getGlobalX();
120 		});
121 		
122 		/**
123 		* mouse y 座標を返します.
124 		* @return {Number}
125 		*/
126 		def(function getMouseY() {
127 			return app.mouseX.mouseY - this.getGlobalY();	
128 		});
129 		
130 		/**
131 		* 表示領域左上隅からの x 座標を返します.
132 		* @return {Number}
133 		*/
134 		def(function getGlobalX() {
135 			var tmpX = 0;
136 			tmpX = this.domElement.offsetLeft;
137 			var pa = this.domElement.offsetParent;
138 			while (pa) {
139 				tmpX += pa.offsetLeft;
140 				pa = pa.offsetParent;
141 			}
142 			return tmpX;
143 		});
144 		
145 		/**
146 		* 表示領域左上隅からの y 座標を返します.
147 		* @return {Number}
148 		*/
149 		def(function getGlobalY() {
150 			var tmpY = 0;
151 			tmpY = this.domElement.offsetTop;
152 			var pa = this.domElement.offsetParent;
153 			while (pa) {
154 				tmpY += pa.offsetTop;
155 				pa = pa.offsetParent;
156 			}
157 			return tmpY;
158 		});
159 		
160 		/**
161 		* このオブジェクトの矩形領域を返します.
162 		* @return {Rectangle}
163 		*/
164 		def(function getBounds(targetCoordinateSpace) {
165 			var targetXAtGlobal = targetCoordinateSpace.getGlobalX();
166 	 		var targetYAtGlobal = targetCoordinateSpace.getGlobalY();
167 	 		var thisXAtGlobal = this.getGlobalX();
168 	 		var thisYAtGlobal = this.getGlobalY();
169 	 		return nsg.Rectangle.gen(thisXAtGlobal - targetXAtGlobal, thisYAtGlobal  -targetXAtGlobal);
170 		});
171 		
172 		def(function getRect(targetCoordinateSpace) {
173 			return this.getBounds(targetCoordinateSpace);
174 		});
175 		
176 		/**
177 		* 引数に与えられたグローバル座標の、このDisplayObjectでの座標にして返します.
178 		* @param globalP {Point}
179 		* @return {Point}
180 		*/
181 		def(function globalToLocal(globalP) {
182 			return nsg.Point.gen(this.x + globalP.x, this.y + globalP.y);
183 		});
184 		
185 		def(function paint() {
186 			var style = this.domElement.style;
187 			var ua = app.userAgent;
188 			if (ua.isIE()) { style.filter = "alpha(opacity=" + this.alpha * 100 + ")"; }
189 			else style.opacity = this.alpha.toString();
190 		});
191 		
192 		def(function rendering() {
193 			this.paint();
194 		});
195 		
196 		/**
197 		* 
198 		*/
199 		def(function removeFromSuperview() {
200 			this.domElement.parentNode.removeChild(this.de);
201 		});
202 	});
203 	
204 	
205 	/** 	
206 	* @class DisplayObjectがユーザー操作を受け付けるよう拡張します. 
207 	* @augments DisplayObject
208 	*/
209 	proto(function InteractiveObject() {
210 		ex(ns.DisplayObject);
211 		
212 		/**
213 		* @memberOf InteractiveObject
214 		*/
215 		init(function(domElement) {
216 			this.domElement = domElement;
217 			this.setEvents();
218 		});
219 		
220 		/**
221 		* マウスポインタがこの InteractiveObject オブジェクト上に重なった時, ハンドカーソルに変更するかどうか判別します.
222 		* @return {Boolean}
223 		*/
224 		def(function getUseHandCursor() {
225 			return this.domElement.style.cursor == "pointer";
226 		});
227 		
228 		/**
229 		* マウスポインタがこの InteractiveObject オブジェクト上に重なった時, ハンドカーソルに変更するかどうか指定します.
230 		* @return {Boolean}
231 		*/
232 		def(function setUseHandCursor(val) {
233 			if (val) {
234 				this.domElement.style.cursor = "pointer";
235 			}
236 			else {
237 				this.domElement.style.cursor = "default";
238 			}
239 		});
240 		
241 		/** @private */
242 		def(function setEvents() {
243 			var self = this;
244 			var g = nsc.Utilitie.gen();
245 			var isOvered = false;
246 			g.listen(this.domElement, nse.DOMMouseEvent.MOUSE_OVER, function(evt){
247 				if (isOvered == false) {
248 					isOvered = true;
249 					self.dispatchEvent(nse.FLMouseEvent.gen(nse.FLMouseEvent.ROLL_OVER, self));
250 				}
251 			});
252 			g.listen(this.domElement, nse.DOMMouseEvent.MOUSE_OUT, function(evt){
253 				if (isOvered) {
254 					var rect = nsg.Rectangle.gen(self.getGlobalX(), self.getGlobalY(), self.getWidth(), self.getHeight());
255 					if (isOvered) {
256 						isOvered = false;
257 						self.dispatchEvent(nse.FLMouseEvent.gen(nse.FLMouseEvent.ROLL_OUT, self, evt));
258 					}
259 					// 1フレまつ
260 					/*
261 					new Wait(1, function() {
262 						if (!rect.contains(Application.mouseX, Application.mouseY)) {
263 							if (isOvered) {
264 								isOvered = false;
265 								self.dispatchEvent(new nse.FLMouseEvent(nse.FLMouseEvent.ROLL_OUT, self, evt));
266 							}
267 						}
268 					})*/
269 				}
270 			});
271 			g.listen(this.domElement, nse.DOMMouseEvent.MOUSE_OVER, function(evt){
272 				var mOver = nse.FLMouseEvent.gen(nse.FLMouseEvent.MOUSE_OVER, self, evt);
273 				mOver.target = evt.target || evt.srcElement;
274 				self.dispatchEvent(mOver);
275 			});
276 			g.listen(this.domElement, nse.DOMMouseEvent.MOUSE_OUT, function(evt){
277 				var mOut = nse.FLMouseEvent.gen(nse.FLMouseEvent.MOUSE_OUT, self, evt);
278 				mOut.target = evt.target || evt.srcElement;
279 				self.dispatchEvent(mOut);
280 			});
281 			g.listen(this.domElement, nse.DOMMouseEvent.CLICK, function(evt){
282 				var click = nse.FLMouseEvent.gen(nse.FLMouseEvent.CLICK, self, evt);
283 				click.target = evt.target || evt.srcElement;
284 				self.dispatchEvent(click);
285 			});
286 			g.listen(this.domElement, nse.DOMMouseEvent.MOUSE_DOWN, function(evt){
287 				var down = nse.FLMouseEvent.gen(nse.FLMouseEvent.MOUSE_DOWN, self, evt);
288 				down.target = evt.target || evt.srcElement;
289 				self.dispatchEvent(down);
290 			});
291 			g.listen(this.domElement, nse.DOMMouseEvent.MOUSE_UP, function(evt){
292 				var up = nse.FLMouseEvent.gen(nse.FLMouseEvent.MOUSE_UP, self, evt);
293 				up.target = evt.target || evt.srcElement;
294 				self.dispatchEvent(up);
295 			});
296 		});
297 		/*
298 		def(function paint() {
299 			var style = self.domElement.style;
300 			var ua = Application.userAgent;
301 			if (ua.isIE()) style.filter = "alpha(opacity=" + self.alpha * 100 + ")";
302 			else style.opacity = self.alpha.toString();
303 		});*/
304 	});
305 });
