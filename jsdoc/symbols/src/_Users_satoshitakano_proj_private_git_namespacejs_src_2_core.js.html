 59 	 /** 
 60 	 * @class 通知センター. 多対多のイベント通知構造を表します.
 61 	 */
 62 	 singleton(function NotificationCenter() {
 63 	 	$$.targets = {};
 64 	 	
 65 	 	/** Observer を追加します.
 66 	 	* @param target {Object} 任意の型の Observer.
 67 	 	* @param func {Function} イベント通知を受け取る関数.
 68 	 	* @param notifName {String} 受け取る通知名.
 69 	 	* @memberOf NotificationCenter
 70 	 	*/
 71 	 	$$.addObserver = function(target, func, notifName) {
 72 			var obj = {target:target, func:func};
 73 			if (!this.targets.hasOwnProperty(notifName))
 74 				this.targets[notifName] = [];
 75 			var list = this.targets[notifName];
 76 			var isAlreadyPushed = false;
 77 			var l = list.length;
 78 			for (var i = 0; i < l; i++)
 79 			{
 80 				if (target == list[i].target)
 81 				{
 82 					isAlreadyPushed = true;
 83 					break;
 84 				}
 85 			}
 86 			if (!isAlreadyPushed)
 87 				list.push(obj);
 88 		}
 89 		
 90 		/** 
 91 		* Observer を削除します.
 92 	 	* @memberOf NotificationCenter
 93 		*/
 94 		$$.removeObserver = function(target, notifName) {
 95 			var list = this.targets[notifName];
 96 			var l = list.length;
 97 			for (var i = 0; i < l; i++) {
 98 				if (target == list[i].target) {
 99 					list.splice(i, 1);
100 					return;
101 				}
102 			}
103 		}
104 		
105 		/**
106 		* 通知を発行します.
107 		* @param notification {Notification} 通知オブジェクト.
108 	 	* @memberOf NotificationCenter
109 		*/
110 		$$.postNotification = function(notification) {
111 			var list = this.targets[notification.name];
112 			
113 			if (list == null)
114 				return;
115 			var l = list.length;
116 			for (var i = 0; i < l; i++)
117 			{
118 				var obj = list[i];
119 				obj.func.call(obj.target, notification.object);
120 			}
121 		}
122 	 })

146 	 /** 
147 	 * @class ユニークな識別子.
148 	 */
149 	 proto(function Identifier() {
150 		 /** @memberOf Identifier */
151 	 	init(function() {
152 	 		this.id = null;
153 	 		generate.call(this);
154 	 	});
155 	 	
156 		$$.IDs = [];
157 	 	
158 	 	/** @private */
159 	 	function generate() {
160 	 		this.id = (new Date()).getTime();
161 			while (!check.call(this))
162 				this.id = Math.round(this.id * Math.random());
163 			this.$class.IDs.push(this.id);
164 	 	}
165 	 	
166 	 	/** @private */
167 	 	function check() {
168 	 		var numIDs = this.$class.IDs.length;
169 			var res = true;
170 			for (var i = 0; i < numIDs; i++) {
171 				if (this.$class.IDs[i] == this.id)
172 					res = false;
173 			}
174 			return res;
175 	 	}
176 	 });

201 	/**
202 	* @class 数値の範囲を表します.
203 	*/
204 	proto(function Range() {
205 		/**
206 		* @memberOf Range
207 		* @min {Number} 範囲の最小値
208 		* @max {Number} 範囲の最大値
209 		*/
210 		init(function(min, max) {
211 			this.min = min;
212 			this.max = max;
213 		});
214 		
215 		/**
216 		* 引数に渡された Range オブジェクトが、自身の値の範囲内に収まる場合に true を返します.
217 		* @param {Range} 評価対象の Range オブジェクト
218 		* @return {Boolean}
219 		*/
220 		def(function contains(range) {
221 			return (this.min <= range.min && range.max <= this.max);
222 		});
223 		
224 		/**
225 		* 値を別の Range での値に置き換えます.
226 		* @description 例えば 0 ~ 100 の Range オブジェクトに対して,<br/>
227 		* remap(50, Range.gen(100, 200)); を評価した場合,<br/>
228 		* 150 が返されます.
229 		* @param value {Number} この範囲内での１点の値.
230 		* @param range {Range} 置き換える範囲
231 		* @return
232 		*/
233 		def(function remap(value, range) {
234 			return range.min + (range.length()) * this.ratio(value);
235 		})
236 		
237 		/** @return {Number} 範囲の大きさを返します. */
238 		def(function length() {
239 			return this.max - this.min;
240 		})
241 		
242 		/** 
243 		* 引数に渡された値の、この範囲内での割合を返します.
244 		* @param val {Number} 評価する値.
245 		* @return {Number} 
246 		*/
247 		def(function ratio (val) {
248 			return (val - this.min) / this.length();
249 		})
250 		
251 	});

257 	proto(function Operation() {
258 		/**
259 		* @memberOf Operation
260 		* @param scope {Object} 関数評価時のスコープ.
261 		* @param func {Function} 操作を表す関数.
262 		* @param args {Array} 関数評価時に引数として渡される arguments 配列. *関数評価時は可変長引数として渡されます.
263 		*/
264 		init(function(scope, func, args) {
265 			this.scope = scope;
266 			this.func = func;
267 			this.args = args;
268 		})
269 		
270 		/** 評価します */
271 		def(function execute() {
272 			this.func.apply(this.scope, this.args);
273 		})
274 		
275 	})

278 	* @class 操作シーケンススタック.
279 	* 
280 	**/
281 	proto(function OperationQueue() {
282 		/** @memberOf OperationQueue */
283 		init(function() {
284 			this.operations = [];
285 		})
286 		
287 		/** 
288 		* 操作を追加します.
289 		* @param op {Operation} Operation オブジェクト.
290 		*/
291 		def(function push(op) {
292 			this.operations.push(op);
293 		})
294 		
295 		/** 
296 		* スタックの最後尾にある Operation オブジェクトを削除します.
297 		* @return {Operation} 削除した Operation オブジェクト.
298 		*/
299 		def(function pop() {
300 			return this.operations.pop();
301 		})
302 		
303 		/**
304 		* 一連の操作シーケンスをスタックに追加された順で実行します.
305 		*/
306 		def(function execute() {
307 			this.operations.each(function (op) {
308 				op.execute();
309 			})
310 		})
311 	})

326 	/**
327 	* @class 操作を記録できるAPIを提供します.
328 	* @description オブジェクトにモジュールとして include して使用します.
329 	**/
330 	proto(function Recordable() {
331 		init(function() {
332 			this.opq = ns.OperationQueue.gen();
333 		})
334 		
335 		/** 
336 		* 操作を記録します.
337 		* @param op {Operation} 記録する Operation. 省略した場合は rec() を呼び出した関数が記録されます.<br/>
338 		* Recordable をモジュールとして include している場合は this.rec() と呼びだすことで<br/>
339 		* 現在居るスコープの関数が操作として記録されます.
340 		*/
341 		def(function rec(op) {
342 			if (!op) {
343 				var caller = arguments.callee.caller;
344 				op = ns.Operation.gen(this, caller, caller.arguments);
345 			}
346 			this.opq.push(op);
347 		})
348 		
349 		/** 
350 		* 記録した一連の操作を順に実行します.
351 		*/
352 		def(function playback() {
353 			this.opq.execute();
354 		})
355 		
356 		/**
357 		* 記録した操作をすべて消去します.
358 		*/
359 		def(function clear() {
360 			this.opq = null;
361 		})
362 	})
