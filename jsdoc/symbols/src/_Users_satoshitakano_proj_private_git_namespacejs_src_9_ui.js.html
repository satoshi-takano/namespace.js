 28 new Namespace(namespace_lib_ui).use(function() {
 29 	var ns = this;
 30 	var nsd = new Namespace(namespace_lib_display);
 31 	
 32 	proto(function Label() {
 33 		ex(nsd.DisplayObject);
 34 		
 35 		init(function(textElement) {
 36 			ns.Label.superClass.initialize.call(this, textElement);
 37 			this.hasInnerHTML = true;
 38 			if (textElement.innerHTML == undefined)
 39 				this.hasInnerHTML = false;
 40 			this.text;
 41 			if(this.hasInnerHTML)
 42 				this.text = textElement.innerHTML;
 43 			else
 44 				this.text = textElement.textContent;
 45 		});
 46 		
 47 		def(function rendering() {
 48 			var box = document.createElement("div");
 49 				
 50 			box.style.position = "absolute";
 51 			var clone = this.domElement.cloneNode(true);
 52 			box.appendChild(clone);
 53 			document.body.appendChild(box);
 54 			var t = this.text;
 55 			clone.style.fontSize = this.domElement.style.fontSize;
 56 
 57 			this.setCloneText(t, clone);
 58 			
 59 			box.style.width = this.targetWidth + "px";
 60 
 61 			while (box.offsetHeight > this.targetHeight) {
 62 				if (t.length == 3) {
 63 					this.setCloneText(t.substring(0, 1), clone);
 64 					return;
 65 				}
 66 				
 67 				t = t.substr(0, t.length-2) + "…";
 68 				this.setCloneText(t, clone);
 69 			}
 70 			
 71 			box.style.width = "";
 72 			while (box.offsetWidth > this.targetWidth) {
 73 				if (t.length == 4) {
 74 					this.setCloneText(t.substring(0, 1), clone);
 75 					return;
 76 				}
 77 				t = t.substr(0, t.length-4) + "...";
 78 				this.setCloneText(t, clone);
 79 			}
 80 
 81 			document.body.removeChild(box)
 82 			this.setText(t);
 83 		});
 84 		
 85 		def(function setText(val) {
 86 			if (this.hasInnerHTML)
 87 				this.domElement.innerHTML = val;
 88 			else
 89 				this.domElement.textContent = val;
 90 		});
 91 		
 92 		def(function setCloneText(val, clone) {
 93 			if (clone.innerHTML == undefined)
 94 				clone.textContent = val;
 95 			else
 96 				clone.innerHTML = val;
 97 		});
 98 		
 99 		def(function getText() {
100 			if (this.hasInnerHTML) 
101 				return this.domElement.innerHTML;
102 			else
103 				return this.domElement.textContent;
104 		});
105 		
106 		def(function setWidth(val) {
107 			this.targetWidth = val;
108 		});
109 		
110 		def(function setHeight(val) {
111 			this.targetHeight = val;
112 		});
113 	});
114 });
