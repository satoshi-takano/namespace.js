 32 	 /** 
 33 	 * @class 点を表します.
 34 	 */
 35 	proto(function Point() {
 36 		/**
 37 		* @memberOf Point
 38 		* @param x {Number} x 座標
 39 		* @param y {Number} y 座標　
 40 		*/
 41 		init(function(x, y) {
 42 			this.x = x;
 43 			this.y = y;
 44 		});
 45 		
 46 		/** 
 47 		* このPointを引数のPoint分移動します.
 48 		* @param p {Point} 
 49 		*/
 50 		def(function add(p) {
 51 			return ns.Point.gen(this.x + p.x, this.y + p.y);
 52 		});
 53 		
 54 		/** 
 55 		* Point オブジェクトのクローンを返します.
 56 		* @return {Point} clone
 57 		*/
 58 		def(function clone() {
 59 			return ns.Point.gen(this.x, this.y);
 60 		});
 61 		
 62 		/**
 63 		* ２点の差をスカラー値で返します.
 64 		* @return {Number}
 65 		*/
 66 		def(function distance(p1, p2) {
 67 			var dx = p1.x - p2.x;
 68 			var dy = p1.y - p2.y;
 69 			return Math.abs(Math.sqrt(dx*dx + dy*dy));
 70 		});
 71 		
 72 		/** 
 73 		* ２点が等しいかどうか評価します.
 74 		* @return {Boolean}
 75 		*/
 76 		def(function equals(p) {
 77 			return this.x == p.x && this.y == p.y;
 78 		});
 79 		
 80 		/**
 81 		* Pointの x, y 座標それぞれにオフセット分加算します.
 82 		* @param dx {Number}
 83 		* @param dy {Number}
 84 		*/
 85 		def(function offset(dx, dy) {
 86 			this.x += dx;
 87 			this.y += dy;
 88 		});
 89 		
 90 		/** 
 91 		* 引数に渡された Point 分を差し引いた新しい Point オブジェクトを返します.<br/>
 92 		* 現在の Point オブジェクトの値は変更されません.
 93 		* @param p {Point}
 94 		*/
 95 		def(function subtract(p) {
 96 			return ns.Point.gen(this.x - p.x, this.y - p.y);
 97 		});
 98 		
 99 		/**
100 		* 原点を (0, 0) とした場合のこの Point オブジェクトの原点からの距離を返します.
101 		* @return {Number}
102 		*/
103 		def(function getLength() {
104 			return Math.sqrt(this.x*this.x + this.y*this.y);
105 		});
106 		
107 		/**
108 		* ２点間を比 f (0 ~ 1) で線形補間します.
109 		* @param p1 {Point} 
110 		* @param p2 {Point}
111 		* @parma f {Number}
112 		* @return {Point} p1, p2 の割合 f に位置する新しい Point オブジェクト
113 		* @memberOf Point
114 		*/
115 		$$.def(function interpolate(p1, p2, f) {
116 			var dx = p2.x - p1.x;
117 			var dy = p2.y - p1.y;
118 			var tx = p1.x + dx * (1-f);
119 			var ty = p1.y + dy * (1-f);
120 			return ns.Point.gen(tx, ty);
121 		});
122 		
123 		/** 
124 		* 極座標から直交座標に変換します.
125 		* @memberOf Point
126 		*/
127 		$$.def(function polar(length, radian) {
128 			return ns.Point.gen(length * Math.cos(radian), length * Math.sin(radian));
129 		});
130 	});

133 	 /** 
134 	 * @class 矩形を表します.
135 	 */
136 	 proto(function Rectangle() {
137 		 /**
138 		 * @param x {Number} x 座標
139 		 * @param y {Number} y 座標
140 		 * @param w {Number} 幅
141 		 * @param h {Number} 高さ
142 		 * @memberOf Rectangle
143 		 */
144 	 	init(function (x, y, w, h) {
145 	 		this.x = x;
146 	 		this.y = y;
147 	 		this.width = w;
148 	 		this.height = h;
149 	 	});
150 		
151 		/**
152 		* Rectangleオブジェクトのクローンを返します.
153 		* @return {Rectangle} 
154 		*/
155 		def(function clone() {
156 			return ns.Rectangle.gen(this.x, this.y, this.width, this.height);
157 		});
158 		
159 		/** 
160 		* 指定されたポイントがこの Rectangle オブジェクトで定義される矩形領域内にあるかどうかを判別します。
161 		* @param x {Number} x 座標
162  		* @param y {Number} y 座標
163  		* @return {Boolean}  このオブジェクトに指定されたオブジェクトが含まれる場合は true を返します.
164 		*/
165 		def(function contains(x, y) {
166 			return (this.x < x && this.x + this.width > x) && (this.y < y && this.y + this.height > y);
167 		});
168 		
169 		/** 
170 		* 指定されたポイントがこの Rectangle オブジェクトで定義される矩形領域内にあるかどうかを判別します。
171 		* @param p {Point} Point
172  		* @return {Boolean}  このオブジェクトに指定されたオブジェクトが含まれる場合は true を返します.
173 		*/
174 		def(function containsPoint(p) {
175 			return this.contains(p.x, p.y);
176 		});
177 		
178 		/** 
179 		* 指定された矩形領域がこの Rectangle オブジェクトで定義される矩形領域内にあるかどうかを判別します。
180 		* @param rect {Rectangle} 
181  		* @return {Boolean}  このオブジェクトに指定されたオブジェクトが含まれる場合は true を返します.
182 		*/
183 		def(function containsRect(rect) {
184 			return this.contains(rect.x, rect.y) && this.contains(rect.x + rect.width, rect.y + rect.height);
185 		});
186 		
187 		/**
188 		* 引数に与えられた Rectangle オブジェクトと等しいかどうか判別します.
189 		* @param rect {Rectangle} 
190 		* @return {Boolean} 等しい場合には true を返します.
191 		*/
192 		def(function equals(rect) {
193 			return (this.x == rect.x && this.y == rect.y && this.width == rect.width && this.height == rect.height);
194 		});
195 		
196 		/** 
197 		* 幅、高さを拡張します.
198 		* @param dx {Number} offset x
199 		* @param dy {Number} offset y
200 		*/
201 		def(function inflate(dx, dy) {
202 			this.width += dx;
203 			this.height += dy;
204 		});
205 		
206 		/**
207 		* 幅、高さを拡張します.
208 		* @param p {Point} offsets
209 		*/
210 		def(function inflatePoint(p) {
211 			this.width += p.x;
212 			this.height += p.y;
213 		});
214 		
215 		/**
216 		* 引数に与えられた Rectangle オブジェクトと交差する領域を新しい Rectangle オブジェクトとして返します.
217 		* @return {Rectangle}
218 		*/
219 		def(function intersection(rect) {
220 			var r = ns.Rectangle.gen(0, 0, 0, 0);
221 			r.x = Math.max(this.x, rect.x);
222 			r.y = Math.max(this.y, rect.y);
223 			var w;
224 			if (this.getRight() < rect.getRight())
225 				w = this.getRight() - rect.x;
226 			else
227 				w = rect.getRight() - this.x;
228 			var h;
229 			if (this.getBottom() < rect.getBottom())
230 				h = this.getBottom() - rect.y;
231 			else
232 				h = rect.getBottom() - this.y;
233 			r.width = w;
234 			r.height = h;
235 			if (0 < r.width && 0 < r.height)
236 				return r;
237 			else
238 				return null;
239 		});
240 		
241 		/**
242 		* 引数に与えられた Rectangle オブジェクトと交差する領域があるかどうか判別します.
243 		* @param {Rectangle} 
244 		* @return 交差する領域がある場合に true を返します.
245 		*/
246 		def(function intersect(rect) {
247 			if (this.intersection)
248 				return true;
249 			else
250 				return false;
251 		});
252 		
253 		/**
254 		* 矩形を移動します.
255 		* @param dx {Number} offset x
256 		* @param dy {Number} offset y
257 		*/
258 		def(function offset(dx, dy) {
259 			this.x += dx;
260 			this.y += dy;
261 		});
262 		
263 		/**
264 		* 矩形を移動します.
265 		* @param dx {Point} offsets
266 		*/
267 		def(function offsetPoint() {
268 			this.x += p.x;
269 			this.y += p.y;
270 		});
271 		
272 		/**
273 		* ２つの矩形領域を合成します.
274 		* @param rect {Rectangle} 
275 		*/
276 		def(function union(rect) {
277 			var intersection = this.intersection(rect);
278 			var r = ns.Rectangle.gen(0,0,0,0);
279 			r.x = Math.min(this.x, rect.x);
280 			r.y = Math.min(this.y, rect.y);
281 			r.width = this.width + rect.width - intersection.width;
282 			r.height = this.height + rect.height - intersection.height;
283 			return r;
284 		});
285 		
286 		/**
287 		* 底辺の y 座標を返します.
288 		* @return {Number}
289 		*/
290 		def(function getBottom() {
291 			return this.y + this.height;
292 		});
293 		
294 		/** 
295 		* 右下隅の座標を返します.
296 		* @return {Point}
297 		*/
298 		def(function getBottomRight() {
299 			return ns.Point.gen(this.getRight(), this.getBottom());
300 		});
301 		
302 		/**
303 		* 右辺の x 座標を返します.
304 		* @return {Number}
305 		*/
306 		def(function getRight() {
307 			return this.x + this.width;
308 		});
309 		
310 		/**
311 		* サイズを返します.
312 		* @return {Point}
313 		*/
314 		def(function getSize() {
315 			return ns.Point.gen(this.width, this.height);
316 		});
317 		
318 		/**
319 		* 左上隅の座標を返します.
320 		* @return {Point}
321 		*/
322 		def(function getTopLeft() {
323 			return ns.Point.gen(this.x, this.y);
324 		});
325 	 });

327 	 /** 
328 	 * @class 3 * 3 行列
329 	 */
330 	 proto(function Matrix() {
331 	 	init(function (a, b, c, d, tx, ty) {
332 	 		this.a = a;
333 	 		this.b = b;
334 	 		this.c = c;
335 	 		this.d = d;
336 	 		this.tx = tx;
337 	 		this.ty = ty;
338 	 	});
339 	 	
340 	 	/**
341 	 	* クローンを返します.
342 	 	* @return {Matrix}
343 	 	*/
344 	 	def(function clone() {
345 		 	return ns.Matrix.gen(this.a, this.b, this.c, this.d, this.tx, this.ty);
346 	 	});
347 	 	
348 	 	/**
349 	 	* 乗算します.
350 	 	* @param m {Matrix}
351 	 	*/
352 	 	def(function cancat(m) {
353 	 		var a = this.a, b = this.b, c = this.c, d = this.d, tx = this.tx, ty = this.ty;	
354 			this.a = a * m.a + c * m.b;
355 			this.c = a * m.c + c * m.d;
356 			this.tx = a * m.tx + c * m.ty + tx;
357 			this.b = b * m.a + d * m.b;
358 			this.d = b * m.c + d * m.d;	
359 			this.ty = b * m.tx + d * m.ty + ty;
360 	 	});
361 	 	
362 	 	/**
363 	 	* 単位行列にします
364 	 	*/
365 	 	def(function identity() {
366 	 		this.a = this.d = 1;
367 			this.b = this.c = this.tx = this.ty = 0;
368 	 	});
369 	 	
370 	 	/** 
371 	 	* 回転させます.
372 	 	* @param radian {Number} 
373 	 	*/
374 	 	def(function rotate(radian) {
375 	 		var rot = ns.Matrix.gen(0, 0, 0, 0, 0, 0);
376 	 		var s = Math.sin(radian);
377 	 		var c = Math.cos(radian);
378 			rot.a = c;
379 			rot.b = s;
380 			rot.c = -s;
381 			rot.d = c;
382 			this.concat(rot);
383 	 	});
384 	 	
385 	 	/**
386 	 	* 伸縮します.
387 	 	* @param scl {Number}
388 	 	*/
389 	 	def(function scale(scl) {
390 	 		var s = ns.Matrix.gen(scl, 0, 0, scl, 0, 0);
391 			this.concat(s);
392 	 	});
393 	 	
394 	 	/**
395 	 	* 平行移動します.
396 	 	* @param dx {Number}
397 	 	* @param dy {Number}
398 	 	*/
399 	 	def(function translate(dx, dy) {
400 	 		this.tx += dx;
401 			this.ty += dy;
402 	 	});
403 	 });
