 33 	* @class easing 関数を使用して補間した値を配列に保持します.
 34 	*/
 35 	proto(function Interpolator() {
 36 		/**
 37 		* @param to {Number} 終点
 38 		* @parma from {Number} 始点
 39 		* @param easing {} イージング関数オブジェクト
 40 		* @memberOf Interpolator
 41 		*/
 42 		init(function(to, from, step, easing) {
 43 			this.values = [];
 44 			var t = 0;
 45 			var c = to - from;
 46 			while (t++ < step) {
 47 				var v = easing.call(this, t, from, c, step);
 48 				this.values.push(v);
 49 			}
 50 		});
 51 	});

 54 	/** 	
 55 	* @class オブジェクトのパラメータを easing 補間するための高レベル API 
 56 	*/
 57 	proto(function Animator() {
 58 		/** 
 59 		* @memberOf Animator
 60 		* @param delay {Number} 遅延時間 (msec unit)
 61 		*/
 62 		init(function (delay) {
 63 			this.delay = delay;
 64 			this.next = null;
 65 		});
 66 		
 67 		/**
 68 		* オブジェクトのパラメータを補間します.
 69 		* @param target {Object} 
 70 		* @param style {String} css style
 71 		* @param to {Number} 
 72 		* @param step {Number} number of step
 73 		* @param easing {Object} easing function 
 74 		* @param delay {Number} 遅延時間 (msec unit)
 75 		*/ 
 76 		def(function tween(target, style, to, step, easing, delay) {
 77 			var self = this;
 78 			this.delay += (delay || 0);
 79 			setTimeout(function() {
 80 				var cnt = 0;
 81 				var styleValue = target[style];
 82 				var from = parseInt(styleValue);
 83 				var interp = ns.Interpolator.gen(to, from, step, easing);
 84 				var values = interp.values;
 85 				step = step-1;
 86 				var timerID = setInterval(function(){
 87 					target[style] = values[cnt];
 88 					if (self.upFunc)
 89 						self.upFunc.call();
 90 					if (cnt >= step) {
 91 						clearInterval(timerID);
 92 						if (self.cbFunc)
 93 							self.cbFunc.call()
 94 					}
 95 					if (target.rendering) target.rendering();
 96 					cnt++;
 97 				}, 1000 / 60);
 98 			}, this.delay);
 99 			this.next = ns.Animator.gen(this.delay + 1000 / 60 * step);
100 			
101 			return this;
102 		});
103 	});

114 		/** 
115 		* easeIn interpolation
116 		* @param t {Number} iterator
117 		* @param b {Number} Starting Point.
118 		* @param c {Number} offset length.
119 		* @param d {Number} number of total steps
120 		* @memberOf Bounce
121 		*/
122 		$$.def(function easeIn(t, b, c, d) {
123 			if ((t = (d - t) / d) < (1 / 2.75)) { return c - (c * (7.5625 * t * t)) + b; }
124 			if (t < (2 / 2.75)) { return c - (c * (7.5625 * (t -= (1.5 / 2.75)) * t + 0.75)) + b; }
125  			if (t < (2.5 / 2.75)) { return c - (c * (7.5625 * (t -= (2.25 / 2.75)) * t + 0.9375)) + b; }
126  			return c - (c * (7.5625 * (t -= (2.625 / 2.75)) * t + 0.984375)) + b;
127 		})

129 		/** 
130 		* easeInOut interpolation
131 		* @param t {Number} iterator
132 		* @param b {Number} Starting Point.

134 		* @param d {Number} number of total steps
135 		* @memberOf Bounce
136 		*/
137 		def(function easeOut(t, b, c, d) {
138 				if ((t /= d) < (1 / 2.75)) { return c * (7.5625 * t * t) + b; }
139 				else if (t < (2 / 2.75)) { return c * (7.5625 * (t -= (1.5 / 2.75)) * t + .75) + b; }
140 				else if (t < (2.5 / 2.75)) { return c * (7.5625 * (t -= (2.25 / 2.75)) * t + .9375) + b; }
141 				else { return c * (7.5625 * (t -= (2.625 / 2.75)) * t + .984375) + b; }
142 		})

155 			return c / 2 * ((t -= 2) * t * (((s *= (1.525)) + 1) * t + s) + 2) + b;

259 		/** 
260 		* easeIn interpolation
261 		* @param t {Number} iterator
262 		* @param b {Number} Starting Point.
263 		* @param c {Number} offset length.
264 		* @param d {Number} number of total steps
265 		* @memberOf Elastic
266 		*/
267 		$$.def(function easeIn(t, b, c, d) {
268 			var a, p;
269 			if (t == 0)
270 				return b;
271 			if ((t /= d) == 1)
272 				return b + c;
273 			if (!p)
274 				p = d * .3;
275 			if (!a || a < Math.abs(c))
276 			{
277 				a = c;
278 				var s = p / 4;
279 			}
280 			else
281 				var s = p / (2 * Math.PI) * Math.asin(c / a);
282 			return -(a * Math.pow(2, 10 * (t -= 1)) * Math.sin((t * d - s) * (2 * Math.PI) / p)) + b;
283 		})
284 		
285 		/** 
286 		* easeOut interpolation
287 		* @param t {Number} iterator
288 		* @param b {Number} Starting Point.
289 		* @param c {Number} offset length.
290 		* @param d {Number} number of total steps
291 		* @memberOf Elastic
292 		*/
293 		$$.def(function easeOut(t, b, c, d) {
294 			var a, p;
295 			if (t == 0)
296 				return b;
297 			if ((t /= d) == 1)
298 				return b + c;
299 			if (!p)
300 				p = d * .3;
301 			if (!a || a < Math.abs(c))
302 			{
303 				a = c;
304 				var s = p / 4;
305 			}
306 			else
307 				var s = p / (2 * Math.PI) * Math.asin(c / a);
308 			return (a * Math.pow(2, -10 * t) * Math.sin((t * d - s) * (2 * Math.PI) / p) + c + b);
309 		})
310 		
311 		/** 
312 		* easeInOut interpolation
313 		* @param t {Number} iterator
314 		* @param b {Number} Starting Point.
315 		* @param c {Number} offset length.
316 		* @param d {Number} number of total steps
317 		* @memberOf Elastic
318 		*/
319 		$$.def(function easeInOut(t, b, c, d) {
320 			var a, p;
321 			if (t == 0)
322 				return b;
323 			if ((t /= d / 2) == 2)
324 				return b + c;
325 			if (!p)
326 				p = d * (.3 * 1.5);
327 			if (!a || a < Math.abs(c))
328 			{
329 				a = c;
330 				var s = p / 4;
331 			}
332 			else
333 				var s = p / (2 * Math.PI) * Math.asin(c / a);
334 			if (t < 1)
335 				return -.5 * (a * Math.pow(2, 10 * (t -= 1)) * Math.sin((t * d - s) * (2 * Math.PI) / p)) + b;
336 			return a * Math.pow(2, -10 * (t -= 1)) * Math.sin((t * d - s) * (2 * Math.PI) / p) * .5 + c + b;
337 		})
338 	})
