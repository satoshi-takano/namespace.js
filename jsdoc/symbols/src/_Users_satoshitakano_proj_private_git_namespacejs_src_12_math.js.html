 45 		* @return {Number}
 46 		* @memberOf Math
 47 		*/
 48 		$$.factorial = function(n) {
 49 			if (1 < n) return n * ns.Math.factorial(n - 1)

108 		/**
109 		* @class 離散フーリエ変換
110 		* 
111 		**/
112 		proto(function DFT() {
113 			/** 
114 			* @param N {Number} DFT size
115 			* @memberOf DFT 
116 			*/
117 			init(function(N) {
118 				this.N = N;
119 			})
120 			
121 			/**
122 			* 離散フーリエ変換　します.
123 			* @description 引数の x, y には実行後、周波数領域での値が入っています.
124 			* @param x {Array} 実数の入った配列.
125 			* @param y {Array} 虚数の入った配列.
126 			*/
127 			def(function dft(x, y) {
128 				var dt = this.N;
129 				var calcR = [];
130 				var calcI = [];
131 				for (var i = 0; i < dt; i++) {
132 					calcR[i] = 0;
133 					calcI[i] = 0;
134 					var fn = i / dt;
135 					var pi2ft = 2 * M.PI * fn;
136 					for (var t = 0; t < dt; t++) {
137 						calcR[i] += x[t] * M.cos(pi2ft * t);
138 						calcI[i] -= x[t] * M.sin(pi2ft * t);
139 					}
140 				}
141 				for (i = 0; i < dt; i++) {
142 					x[i] = calcR[i];
143 					y[i] = calcI[i];
144 				}
145 			})
146 		})

149 		/**
150 		* @class 高速フーリエ変換
151 		**/
152 		proto(function FFT() {
153 			/**
154 			* @param N {Number} FFT size
155 			* @memberOf FFT
156 			*/
157 			init(function(N) {
158 				this.N = N;
159 				
160  			 	this.bitrev = [];
161 	 			betrev(N, this.bitrev);
162 				this.sintable = [];
163 				sintbl(N, this.sintable);
164 			})
165 			
166 			/**
167 			* 高速フーリエ変換します.
168 			* @description 引数の x, y には実行後、周波数領域での値が入っています.
169 			* @param x {Array} 実数の入った配列.
170 			* @param y {Array} 虚数の入った配列.
171 			*/
172 			def(function fft(x, y) {
173 				last_n = 0;
174 				var bitrev = this.bitrev;
175 				var sintbl = this.sintable;
176 				var n = this.N;
177 				
178 				var i, j, k, ik, h, d, k2, n4, inverse,
179 				t, s, c, dx, dy;
180 				
181 				if (n < 0) {
182 					n = -n; 
183 					inverse = 1;
184 				} else inverse = 0;
185 				
186 				n4 = n >> 2;
187 				
188 				for (i = 0; i < n; i++) {
189 					j = bitrev[i];
190 					if (i < j) {
191 						t = x[i]; x[i] = x[j]; x[j] = t;
192 						t = y[i]; y[i] = y[j]; y[j] = t;
193 					}
194 				}
195 				for (k = 1; k < n; k = k2) {
196 					h = 0; k2 = k + k; d = n / k2;
197 					for (j = 0; j < k; j++) {
198 						c = sintbl[h + n4];
199 						if (inverse) s = -sintbl[h];
200 						else s = sintbl[h];
201 						for (i = j; i < n; i += k2) {
202 							ik = i + k;
203 							dx = s * y[ik] + c * x[ik];
204 							dy = c * y[ik] - s * x[ik];
205 							x[ik] = x[i] - dx; x[i] += dx;
206 							y[ik] = y[i] - dy; y[i] += dy;
207 						}
208 						h += d;
209 					}
210 				}
211 			})
212 			
213 			
214 			function betrev(N, table) {
215 				var i = 0, j = 0, k = 0;
216 				table[0] = 0;
217 				while (++i < N) {
218 					k = N >> 1;
219 					while (k <= j) { j -= k; k >>= 1; }
220 					j += k;
221 					table[i] = j;
222 				}
223 			}
224 			
225 			function sintbl(N, table) {
226 				var i, n2, n4, n8, c, s, dc, ds, t;
227 				n2 = N >> 1;
228 				n4 = N >> 2;
229 				n8 = N >> 3;
230 				t = M.sin(M.PI / N);
231 				dc = 2 * t * t;
232 				ds = M.sqrt(dc * (2 - dc));
233 				t = 2 * dc;
234 				c = table[n4] = 1;
235 				s = table[0] = 0;
236 				for (i = 1; i < n8; i++) {
237 					c -= dc;
238 					dc += t * c;
239 					s += ds;
240 					ds -= t * s;
241 					table[i] = s;
242 					table[n4 - i] = c;
243 				}
244 				if (n8 != 0) table[n8] = M.sqrt(0.5);
245 				for (i = 0; i < n4; i++) table[n2 - i] = table[i];
246 				for (i = 0; i < n2 + n4; i++) table[i + n2] = -table[i];
247 			}
248 		})
249 	})
