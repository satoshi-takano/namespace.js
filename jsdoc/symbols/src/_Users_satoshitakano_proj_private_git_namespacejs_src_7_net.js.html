 25 /**
 26 * @fileOverview ネットワークに関係するオブジェクトが定義されています.
 27 */
 28 new Namespace(namespace_lib_net).use(function () {
 29 	var ns = this;
 30 	
 31 	/** 
 32 	* @class URL
 33 	*/
 34 	proto(function URL() {
 35 		/**
 36 		* 文字列からURLオブジェクトを作成します.
 37 		* @param url {String}
 38 		* @memberOf URL
 39 		*/
 40 		init(function(url) {
 41 			this.url = url;
 42 			this.scheme = 
 43 			this.port = 
 44 			this.user = 
 45 			this.pass = 
 46 			this.path = 
 47 			this.query = 
 48 			this.fragment = "";
 49 			this.generate();
 50 		});
 51 		
 52 		/** 
 53 		* @private
 54 		*/
 55 		def(function generate() {
 56 			var r = this.reg.exec(this.url);
 57 			if (r) {
 58 				var value = "";
 59 				for (var f in this.fields) {
 60 					value = r[this.fields[f]];
 61 					if (value) {
 62 						this[f] = value;
 63 					}
 64 				}
 65 			}
 66 		});
 67 		
 68 		$$.reg = /^((\w+):\/\/)?((\w+):?(\w+)?@)?([^\/\?:]+):?(\d+)?(\/?[^\?#]+)?\??([^#]+)?#?([^#]*)/;
 69 		$$.fields = {
 70 				"scheme": 2,
 71 				"host": 6,
 72 				"port": 7,
 73 				"user": 4,
 74 				"pass": 5,
 75 				"path": 8,
 76 				"query": 9,
 77 				"fragment": 10
 78 		};
 79 		
 80 	});
 81 	
 82 	/** 
 83 	* @class URL request
 84 	*/
 85 	proto(function URLRequest() {
 86 		/**
 87 		* @param urlString {String}
 88 		* @memberOf URLRequest
 89 		*/
 90 		init(function (urlString) {
 91 			this.url = urlString;
 92 			this.httpMethod = "GET";
 93 			this.httpHeader = {};
 94 			this.httpBody = null;
 95 			this.httpHeader.contentType = "application/x-www-form-urlencoded;charset=UTF-8";
 96 		});
 97 	});
 98 	
 99 	
100 	/** 
101 	* @class 非同期でデータを取得します.
102 	* @augments EventDispatcher
103 	*/
104 	proto(function URLLoader() {
105 		var nse = new Namespace(namespace_lib_events);
106 		
107 		ex(nse.EventDispatcher);
108 		
109 		/**
110 		* @memberOf URLLoader
111 		*/
112 		init(function () {
113 			this.currentXHR = null;
114 			this.data = null;
115 			
116 			var self =  this;
117 			this.readyStateCallback = function(e) {
118 				if (self.currentXHR.readyState == ns.URLLoader.DONE) {
119 					self.data = self.currentXHR.responseText;
120 					self.dispatchEvent(nse.FLEvent.gen(nse.FLEvent.COMPLETE, self, e || null));
121 				}
122 			}
123 		});
124 		
125 		/**
126 		* 通信を開始します.
127 		* @param req {URLRequest}
128 		*/
129 		def(function load(req) {
130 			var xhr = this.currentXHR = this.generateXHR();
131 			xhr.onreadystatechange = this.readyStateCallback;
132 			xhr.open(req.httpMethod, req.url, true, null /* password */);
133 			xhr.setRequestHeader("Content-type", req.httpHeader.contentType);
134 			xhr.send(req.httpBody);
135 		});
136 		
137 		def(function generateXHR() {
138 			var xhr;
139 			if (global.XMLHttpRequest != undefined) {
140 				xhr = new XMLHttpRequest();
141 			} else if (global.ActiveXObject){
142 				xhr = new ActiveXObject("Microsoft.XMLHTTP");
143 			}
144 			return xhr;
145 		});
146 		
147 		$$.UNSET = 0;
148 		$$.OPENED = 1;
149 		$$.HEADERS_RECEIVED = 2;
150 		$$.LOADING = 3;
151 		$$.DONE = 4;
152 	});
153 	
154 });
